5- Create an Automobile class with attributes model year, brand and method honk
Create 3 classes that extend the Automobile class  (Car, SportsCar, JunkCar)
Create the compareTo, toString, and equals methods for them
Create  methods  (accelerate, stop, reverse, honk, …)
Create a Tester class where you will put all your Automobiles into an array and test their methods.


class Automobile {
    int modelYear;
    String brand;

    public Automobile(int modelYear, String brand) {
        this.modelYear = modelYear;
        this.brand = brand;
    }

    public void honk() {
        System.out.println(brand + ": honk!");
    }

    public void accelerate() {
        System.out.println(brand + " is accelerating");
    }

    public void stop() {
        System.out.println(brand + " stopped");
    }

    public void reverse() {
        System.out.println(brand + " is reversing");
    }

    public int compareTo(Automobile other) {
        return this.modelYear - other.modelYear;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Automobile) {
            Automobile other = (Automobile) obj;
            return this.modelYear == other.modelYear &&
                   this.brand.equals(other.brand);
        }
        return false;
    }

    public String toString() {
        return modelYear + " " + brand;
    }
}

class Car extends Automobile {
    public Car(int modelYear, String brand) {
        super(modelYear, brand);
    }
}

class SportsCar extends Automobile {
    public SportsCar(int modelYear, String brand) {
        super(modelYear, brand);
    }

    public void accelerate() {
        System.out.println(brand + " is accelerating fast!");
    }
}

class JunkCar extends Automobile {
    public JunkCar(int modelYear, String brand) {
        super(modelYear, brand);
    }

    public void honk() {
        System.out.println(brand + ": *broken honk noise*");
    }
}

public class Main {
    public static void main(String[] args) {
        Automobile[] autos = new Automobile[3];

        autos[0] = new Car(2015, "Toyota");
        autos[1] = new SportsCar(2022, "Ferrari");
        autos[2] = new JunkCar(1998, "OldMobile");

        for (Automobile a : autos) {
            System.out.println(a.toString());
            a.honk();
            a.accelerate();
            a.stop();
            a.reverse();
            System.out.println();
        }

        System.out.println(autos[0].equals(new Car(2015, "Toyota")));
        System.out.println(autos[0].compareTo(autos[2]));
    }
}
